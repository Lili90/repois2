 45      """Helps us get from django's 'language-countryCode' to the datepicker's 'language' if we 
 46      possibly can. 
 47   
 48      If we pass the django 'language_countryCode' through untouched then it might not 
 49      match an exact language string supported by the datepicker and would default to English which 
 50      would be worse than trying to match the language part. 
 51      """ 
 52   
 53      # Catch empty strings in case one sneeks in 
 54      if not language_country_code: 
 55          return 'en' 
 56   
 57      # Check full language & country code against the supported languages as there are dual entries 
 58      # in the list eg. zh-CN (assuming that is a language country code) 
 59      if language_country_code in supported_languages: 
 60          return language_country_code 
 61   
 62      # Grab just the language part and try that 
 63      language = language_country_code.split('-')[0] 
 64      if language in supported_languages: 
 65          return language 
 66   
 67      # Otherwise return English as the default 
 68      return 'en'

154 -def quote(key, value): 

155      """Certain options support string values. We want clients to be able to pass Python strings in 
156      but we need them to be quoted in the output. Unfortunately some of those options also allow 
157      numbers so we type check the value before wrapping it in quotes. 
158      """ 
159   
160      if key in quoted_options and isinstance(value, string_types): 
161          return "'%s'" % value 
162   
163      if key in quoted_bool_options and isinstance(value, bool): 
164          return {True:'true',False:'false'}[value] 
165   
166      return value
